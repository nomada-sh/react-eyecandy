# Card List

import CardList from '../../../components/CardList';

Example:

```tsx
interface Props {
  size: number
}

const App: React.FC = ({ size = 1 }: Props) => {

  const [data, setData] = useState<number[]>([]);

  function setInfo() {
    let count = 1;
    const list = [];
    while(count <= size) {
      list.push(count);
      count++;
    }
    setData(list);
  }

  useEffect(setInfo, [size]);

  return (
    <Row gutter={[16, 16]} justify="center" align="middle">
      {
        data.map(key => (
          <Col xs={24} md={12} key={key}>
            <Card
              hoverable

            >

              />
            </Card>
          </Col>
        ))
      }
    </Row>
  );
};
```
