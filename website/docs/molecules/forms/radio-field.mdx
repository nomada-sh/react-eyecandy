# Radio Field

## Examples

```jsx
import React from 'react';
import { Radio } from 'antd';

export default function App() {
  return (
    <Radio.Group>
      <Radio value={1}>Option 1</Radio>
      <Radio value={2}>Option 2</Radio>
    </Radio.Group>
  );
}
```
